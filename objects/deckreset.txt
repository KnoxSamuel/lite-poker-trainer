class Deck {

    constructor() {
        this.deck = [];
        this.resetDeck();
        this.shuffleDeck();
    }


    resetDeck() {
        this.deck = [];

        const suits = ['h', 's', 'c', 'd'];
        const values = ['A', 2, 3, 4, 5, 6, 7, 8, 9, 'T', 'J', 'Q', 'K'];

        for (let suit in suits) {
            for (let value in values) {
                this.deck.push(`${values[value]}${suits[suit]}`);
            }
        }
    }
